def ==(other)
  return false unless other.is_a?(self.class)

  name == other.name && query_params == other.query_params
end

def to_s
  if valid?
    "#{http_verb.to_s.upcase} #{full_url} with params: #{query_params}"
  else
    "Invalid request for endpoint: #{name}"
  end
end
